def fibonacci(n):
  a0 = 0
  a1 = 1

  for i in range(n):
    fib = a1 + a0
    print(fib)
    # update
    a0 = a1
    a1 = fib

Este devuelve una lista

def fibonacci_list(n):
  a0 = 0
  a1 = 1
  lista = [a0,a1]

  for i in range(1,n):
    fib = lista[i] + lista[i-1]
    lista.append(fib)
  return lista

mis_15 = fibonacci_list(15)
